	@Test

		System.out.println("Enabled on jre8");

	}

	@EnabledOnJre({JRE.JAVA_10,JRE.JAVA_9})
	void enabledOnJava9OrJava10() {
		System.out.println("enabledOnJava9OrJava10");
	}
	
	@Test
	@DisabledOnJre(JRE.JAVA_9)
	void notOnJava9() {
		
		System.out.println("Not on Java 9");
	}

}
